for
--

```bash
/*
 * Programa que muestra los números del 1 al 10
 */
package Estructuras_Control;

/**
 * @author Edith
 */
public class Est_For {
    public static void main(String[] args) {
        for(int i=1; i<=10;i++)
        {
            System.out.println(i + " "); 
        }
    } 
}

```
